let words = [];
let editingId = null;
let activeFilter = 'Все';
let searchQuery = '';

const STORAGE_KEY = 'german-dict-v3';

function loadData() {
  try {
    const stored = localStorage.getItem(STORAGE_KEY);
    if (stored) {
      const parsed = JSON.parse(stored);
      // Merge: keep user edits but add any new initial words missing by id
      const ids = new Set(parsed.map(w => w.id));
      const merged = [...parsed];
      INITIAL_WORDS.forEach(w => { if (!ids.has(w.id)) merged.push(w); });
      words = merged;
    } else {
      words = INITIAL_WORDS;
    }
    saveData();
  } catch(e) {
    words = INITIAL_WORDS;
  }
}

function saveData() {
  try { localStorage.setItem(STORAGE_KEY, JSON.stringify(words)); } catch(e) {}
}

// ─── RENDER ──────────────────────────────────────────────────────────────────

function renderAll() {
  renderTabs();
  renderWords();
  renderStats();
}

function renderStats() {
  document.getElementById('total-count').textContent = words.length;
  document.getElementById('verb-count').textContent = words.filter(w => w.cat === 'Глагол').length;
}

function renderTabs() {
  const container = document.getElementById('cat-tabs');
  const counts = {};
  CATEGORIES.forEach(c => counts[c] = 0);
  counts['Все'] = words.length;
  words.forEach(w => { if (counts[w.cat] !== undefined) counts[w.cat]++; });

  container.innerHTML = CATEGORIES.map(cat => {
    const n = counts[cat];
    return `<button class="cat-tab ${activeFilter===cat?'active':''}" onclick="setFilter('${cat}')">${cat} <span style="opacity:0.6;font-size:10px;">${n}</span></button>`;
  }).join('');
}

function getFiltered() {
  let list = [...words];
  if (activeFilter !== 'Все') list = list.filter(w => w.cat === activeFilter);
  if (searchQuery) {
    const q = searchQuery.toLowerCase();
    list = list.filter(w =>
      w.de.toLowerCase().includes(q) ||
      w.ru.toLowerCase().includes(q) ||
      (w.notes && w.notes.toLowerCase().includes(q))
    );
  }
  const sort = document.getElementById('sort-select').value;
  if (sort === 'de') list.sort((a,b) => a.de.localeCompare(b.de, 'de'));
  else if (sort === 'ru') list.sort((a,b) => a.ru.localeCompare(b.ru, 'ru'));
  else list.sort((a,b) => b.id - a.id);
  return list;
}

function renderWords() {
  const grid = document.getElementById('word-grid');
  const list = getFiltered();

  if (list.length === 0) {
    grid.innerHTML = `<div class="empty-state">
      <div class="empty-state-icon">📖</div>
      <div class="empty-state-text">Слов не найдено</div>
      <div class="empty-state-sub">Попробуйте изменить поиск или добавьте новое слово</div>
    </div>`;
    return;
  }

  grid.innerHTML = list.map(w => renderCard(w)).join('');
}

function renderCard(w) {
  const caseBadgeClass = {
    'Akkusativ': 'case-akkusativ',
    'Dativ': 'case-dativ',
    'Akkusativ / Dativ': 'case-both',
    '—': 'case-none'
  }[w.verbCase] || 'case-none';

  const verbBlock = w.cat === 'Глагол' && w.verbForms ? `
    <div class="verb-forms">
      <div class="verb-forms-title">Формы глагола</div>
      <div class="verb-grid">
        <span class="vf-label">3-е л. ед.ч.</span><span class="vf-value">${esc(w.verbForms.p3 || '—')}</span>
        <span class="vf-label">Перфект</span><span class="vf-value">${esc(w.verbForms.perf || '—')}</span>
        <span class="vf-label">Präteritum</span><span class="vf-value">${esc(w.verbForms.prät || '—')}</span>
      </div>
      ${w.verbCase ? `<span class="case-badge ${caseBadgeClass}">${esc(w.verbCase)}</span>` : ''}
    </div>` : '';

  const phrases = w.phrases && w.phrases.length
    ? `<div class="phrases">
        <div class="phrases-title">Словосочетания</div>
        ${w.phrases.map(p => `
          <div class="phrase-item">
            <span class="phrase-de">${esc(p.de)}</span>
            <span class="phrase-ru">— ${esc(p.ru)}</span>
          </div>`).join('')}
      </div>` : '';

  const notes = w.notes ? `<div class="divider"></div><div style="font-size:12px;color:var(--ink3);font-style:italic;">${esc(w.notes)}</div>` : '';

  return `
  <div class="word-card" id="card-${w.id}">
    <div class="card-actions">
      <button class="btn-edit" onclick="openEdit(${w.id})" title="Редактировать">✏️</button>
      <button class="btn-edit btn-danger" onclick="deleteWord(${w.id})" title="Удалить" style="font-size:12px;">✕</button>
    </div>
    <div class="card-top">
      <div class="card-left">
        <div class="word-de">${esc(w.de)}</div>
        <div class="word-ru">${esc(w.ru)}</div>
      </div>
      <span class="cat-badge cat-${w.cat}">${w.cat}</span>
    </div>
    ${verbBlock}
    ${phrases}
    ${notes}
  </div>`;
}

function esc(s) {
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

// ─── MODAL ───────────────────────────────────────────────────────────────────

function openModal(id = null) {
  editingId = id;
  document.getElementById('modal-title').textContent = id ? 'Редактировать слово' : 'Добавить слово';
  clearForm();

  if (id) {
    const w = words.find(x => x.id === id);
    if (!w) return;
    document.getElementById('f-de').value = w.de;
    document.getElementById('f-ru').value = w.ru;
    document.getElementById('f-cat').value = w.cat;
    if (w.verbForms) {
      document.getElementById('f-p3').value = w.verbForms.p3 || '';
      document.getElementById('f-perf').value = w.verbForms.perf || '';
      document.getElementById('f-prät').value = w.verbForms.prät || '';
    }
    if (w.verbCase) document.getElementById('f-case').value = w.verbCase;
    document.getElementById('f-notes').value = w.notes || '';
    (w.phrases || []).forEach(p => addPhraseRow(p.de, p.ru));
  } else {
    addPhraseRow();
  }

  handleCatChange();
  document.getElementById('modal-overlay').classList.add('open');
}

function openEdit(id) { openModal(id); }

function closeModal() {
  document.getElementById('modal-overlay').classList.remove('open');
  editingId = null;
}

function handleOverlayClick(e) {
  if (e.target === document.getElementById('modal-overlay')) closeModal();
}

function clearForm() {
  ['f-de','f-ru','f-p3','f-perf','f-prät','f-notes'].forEach(id => {
    document.getElementById(id).value = '';
  });
  document.getElementById('f-cat').value = 'Глагол';
  document.getElementById('f-case').value = 'Akkusativ';
  document.getElementById('phrases-editor').innerHTML = '';
}

function handleCatChange() {
  const isVerb = document.getElementById('f-cat').value === 'Глагол';
  document.getElementById('verb-section').classList.toggle('visible', isVerb);
}

function addPhraseRow(de = '', ru = '') {
  const editor = document.getElementById('phrases-editor');
  const div = document.createElement('div');
  div.className = 'phrase-row';
  div.innerHTML = `
    <input type="text" placeholder="Auf Deutsch..." value="${esc(de)}">
    <input type="text" placeholder="На русском..." value="${esc(ru)}">
    <button class="btn-remove-phrase" onclick="this.parentElement.remove()">×</button>`;
  editor.appendChild(div);
}

function saveWord() {
  const de = document.getElementById('f-de').value.trim();
  const ru = document.getElementById('f-ru').value.trim();
  if (!de || !ru) { alert('Заполните немецкое слово и перевод'); return; }

  const cat = document.getElementById('f-cat').value;
  const isVerb = cat === 'Глагол';

  const verbForms = isVerb ? {
    p3: document.getElementById('f-p3').value.trim(),
    perf: document.getElementById('f-perf').value.trim(),
    prät: document.getElementById('f-prät').value.trim(),
  } : null;

  const verbCase = isVerb ? document.getElementById('f-case').value : null;

  const phrases = [];
  document.querySelectorAll('#phrases-editor .phrase-row').forEach(row => {
    const inputs = row.querySelectorAll('input');
    const pDe = inputs[0].value.trim();
    const pRu = inputs[1].value.trim();
    if (pDe || pRu) phrases.push({ de: pDe, ru: pRu });
  });

  const notes = document.getElementById('f-notes').value.trim();

  if (editingId) {
    const idx = words.findIndex(w => w.id === editingId);
    if (idx >= 0) words[idx] = { ...words[idx], de, ru, cat, verbForms, verbCase, phrases, notes };
  } else {
    const newId = words.length ? Math.max(...words.map(w => w.id)) + 1 : 1;
    words.push({ id: newId, de, ru, cat, verbForms, verbCase, phrases, notes });
  }

  saveData();
  closeModal();
  renderAll();
}

function deleteWord(id) {
  if (!confirm('Удалить это слово?')) return;
  words = words.filter(w => w.id !== id);
  saveData();
  renderAll();
}
